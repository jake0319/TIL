```js
// n은 약수가 1과 자기자신인 숫자들의 곱
//소수목록 [2,3,5,7,11...]을 순회하며 나누기 

function solution(n) {
    var answer = [];
    for(let i=2; n!==1; i++){ //n이 1이 아닐때까지 //반복문 
        if(n%i ===0){//조건문
            //계속 반복작업을 시킬 때 
            while(n % i ===0){ // n이 i로 나누어떨어지면, i로 또 나누어지는지 나눌 수 있을만큼 확인하고 
                n/=i;  // 기존에 n을 i로 나눠라 (계속)만족할때까지 
             }
            answer.push(i)//이제 더이상 안나누어지면 i를 추가
        }
    }
    return answer;
}
```
